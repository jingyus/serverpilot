#### 注册流程改造

```typescript
// packages/cloud/src/auth/cloud-register.ts

/**
 * Cloud 用户注册流程
 *
 * 与 Self-Hosted 的区别：
 * - Self-Hosted: 注册 → 创建用户 → 成为 owner
 * - Cloud: 注册 → 创建 Tenant → 创建用户 → 成为 Tenant owner
 */
export async function cloudRegister(data: {
  email: string;
  password: string;
  name?: string;
  companyName?: string;
}) {
  // 1. 验证邮箱唯一性
  const existing = await getUserByEmail(data.email);
  if (existing) {
    throw new Error('Email already registered');
  }

  // 2. 创建 Tenant（以公司名或用户名命名）
  const tenantName = data.companyName || data.name || data.email.split('@')[0];
  const tenantSlug = generateSlug(tenantName); // 唯一 slug（用于子域名）

  const tenant = await createTenant({
    name: tenantName,
    slug: tenantSlug,
    plan: 'free', // 默认 Free 计划
    maxServers: 1,
    maxUsers: 1,
  });

  // 3. 创建用户（关联到 Tenant）
  const user = await createUser({
    email: data.email,
    password: hashPassword(data.password),
    name: data.name,
    tenantId: tenant.id,
    role: 'owner', // Tenant 的 owner
  });

  // 4. 更新 Tenant 的 ownerId
  await updateTenant(tenant.id, { ownerId: user.id });

  // 5. 创建默认团队设置
  await createDefaultTeamSettings(tenant.id);

  // 6. 发送欢迎邮件
  await sendWelcomeEmail(user.email, {
    tenantSlug,
    dashboardUrl: `https://${tenantSlug}.serverpilot.io`, // 子域名
  });

  // 7. 返回 JWT token
  return {
    user,
    tenant,
    tokens: generateTokens(user.id, tenant.id),
  };
}
```

```typescript
// packages/cloud/src/billing/stripe-integration.ts

import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);

/**
 * 创建订阅（用户升级到 Pro/Team/Enterprise）
 */
export async function createSubscription(
  userId: string,
  tenantId: string,
  plan: 'pro' | 'team' | 'enterprise'
) {
  // 1. 创建或获取 Stripe Customer
  const user = await getUser(userId);
  let customerId = user.stripeCustomerId;

  if (!customerId) {
    const customer = await stripe.customers.create({
      email: user.email,
      metadata: { userId, tenantId },
    });
    customerId = customer.id;
    await updateUser(userId, { stripeCustomerId: customerId });
  }

  // 2. 创建订阅
  const priceId = PLAN_PRICE_IDS[plan]; // Stripe Price ID

  const subscription = await stripe.subscriptions.create({
    customer: customerId,
    items: [{ price: priceId }],
    payment_behavior: 'default_incomplete',
    payment_settings: { save_default_payment_method: 'on_subscription' },
    expand: ['latest_invoice.payment_intent'],
  });

  // 3. 存储订阅信息到数据库
  await createSubscriptionRecord({
    tenantId,
    userId,
    plan,
    status: 'incomplete',
    stripeSubscriptionId: subscription.id,
    currentPeriodStart: subscription.current_period_start * 1000,
    currentPeriodEnd: subscription.current_period_end * 1000,
  });

  // 4. 返回 clientSecret（用于前端确认支付）
  const invoice = subscription.latest_invoice as Stripe.Invoice;
  const paymentIntent = invoice.payment_intent as Stripe.PaymentIntent;

  return {
    subscriptionId: subscription.id,
    clientSecret: paymentIntent.client_secret,
  };
}

/**
 * Stripe Webhook 处理
 */
export async function handleStripeWebhook(
  payload: Buffer,
  signature: string
) {
  const event = stripe.webhooks.constructEvent(
    payload,
    signature,
    process.env.STRIPE_WEBHOOK_SECRET!
  );

  switch (event.type) {
    // 订阅激活
    case 'customer.subscription.created':
    case 'customer.subscription.updated':
      await handleSubscriptionUpdated(event.data.object as Stripe.Subscription);
      break;

    // 订阅取消
    case 'customer.subscription.deleted':
      await handleSubscriptionCanceled(event.data.object as Stripe.Subscription);
      break;

    // 支付失败
    case 'invoice.payment_failed':
      await handlePaymentFailed(event.data.object as Stripe.Invoice);
      break;

    // 支付成功
    case 'invoice.payment_succeeded':
      await handlePaymentSucceeded(event.data.object as Stripe.Invoice);
      break;
  }
}

async function handleSubscriptionUpdated(subscription: Stripe.Subscription) {
  const tenantId = subscription.metadata.tenantId;
  const plan = subscription.metadata.plan as 'pro' | 'team' | 'enterprise';

  // 更新 Tenant 计划
  await updateTenant(tenantId, {
    plan,
    maxServers: PLAN_LIMITS[plan].maxServers,
    maxUsers: PLAN_LIMITS[plan].maxUsers,
  });

  // 更新订阅状态
  await updateSubscription(subscription.id, {
    status: subscription.status,
    currentPeriodStart: subscription.current_period_start * 1000,
    currentPeriodEnd: subscription.current_period_end * 1000,
  });
}

async function handleSubscriptionCanceled(subscription: Stripe.Subscription) {
  const tenantId = subscription.metadata.tenantId;

  // 降级到 Free 计划
  await updateTenant(tenantId, {
    plan: 'free',
    maxServers: 1,
    maxUsers: 1,
  });

  // 发送通知邮件
  await sendSubscriptionCanceledEmail(tenantId);
}

const PLAN_PRICE_IDS = {
  pro: 'price_1ProMonthly',
  team: 'price_1TeamMonthly',
  enterprise: 'price_1EnterpriseMonthly',
};

const PLAN_LIMITS = {
  free: { maxServers: 1, maxUsers: 1 },
  pro: { maxServers: 10, maxUsers: 5 },
  team: { maxServers: Infinity, maxUsers: Infinity },
  enterprise: { maxServers: Infinity, maxUsers: Infinity },
};
```

### 3. AI 配额管理（关键！）

**问题**：
- 文档中定义了配额（Free 100 次/月，Pro 2000 次/月）
- 但没有实际的配额检查和限制机制

**需要实现**：

```typescript
// packages/cloud/src/ai/quota-manager.ts

/**
 * AI 配额管理器
 */
export class AIQuotaManager {
  /**
   * 检查用户配额
   */
  async checkQuota(userId: string, tenantId: string): Promise<QuotaCheckResult> {
    const tenant = await getTenant(tenantId);
    const quota = PLAN_QUOTAS[tenant.plan];

    // Free 用户：次数限制
    if (tenant.plan === 'free') {
      const monthlyCount = await this.getMonthlyCallCount(userId);

      if (monthlyCount >= quota.maxCalls) {
        return {
          allowed: false,
          reason: `已达到 Free 计划限制（${quota.maxCalls} 次/月）`,
          remaining: 0,
          upgradeUrl: '/billing?upgrade=pro',
        };
      }

      return {
        allowed: true,
        remaining: quota.maxCalls - monthlyCount,
      };
    }

    // 付费用户：公平使用政策（软限制）
    const monthlyCost = await this.getMonthlyCost(userId);

    if (monthlyCost > quota.softLimit) {
      // 不拒绝请求，但发送警告
      await this.sendUsageWarning(userId, {
        used: monthlyCost,
        limit: quota.softLimit,
      });
    }

    return {
      allowed: true,
      remaining: quota.softLimit - monthlyCost,
    };
  }

  /**
   * 记录 AI 调用
   */
  async trackAICall(userId: string, call: AICallMetrics) {
    await db.insert(ai_usage).values({
      userId,
      tenantId: call.tenantId,
      model: call.model,
      inputTokens: call.inputTokens,
      outputTokens: call.outputTokens,
      cost: this.calculateCost(call),
      timestamp: Date.now(),
    });

    // 实时检查配额（防止滥用）
    await this.checkQuota(userId, call.tenantId);
  }

  /**
   * 获取本月调用次数
   */
  private async getMonthlyCallCount(userId: string): Promise<number> {
    const startOfMonth = new Date();
    startOfMonth.setDate(1);
    startOfMonth.setHours(0, 0, 0, 0);

    const count = await db
      .select({ count: sql<number>`count(*)` })
      .from(ai_usage)
      .where(
        and(
          eq(ai_usage.userId, userId),
          gte(ai_usage.timestamp, startOfMonth.getTime())
        )
      );

    return count[0]?.count || 0;
  }
}

const PLAN_QUOTAS = {
  free: {
    maxCalls: 100, // 硬限制
  },
  pro: {
    maxCalls: 2000, // 软限制（超过发警告，不拒绝）
    softLimit: 50, // $50/月
  },
  team: {
    softLimit: 200, // $200/月
  },
  enterprise: {
    softLimit: 1000, // $1000/月
  },
};
```

// packages/cloud/src/api/middleware/check-ai-quota.ts

export function checkAIQuota() {
  return async (c: Context, next: Next) => {
    const { userId, tenantId } = c.get('auth');

    const quotaManager = getAIQuotaManager();
    const result = await quotaManager.checkQuota(userId, tenantId);

    if (!result.allowed) {
      return c.json({
        error: {
          code: 'QUOTA_EXCEEDED',
          message: result.reason,
          upgradeUrl: result.upgradeUrl,
        },
      }, 429);
    }

    // 将剩余配额信息添加到 response header
    c.header('X-Quota-Remaining', result.remaining.toString());

    await next();
  };
}

// packages/cloud/src/websocket/cloud-agent-auth.ts

/**
 * Cloud Agent 认证（增强版）
 *
 * 增加 Tenant 隔离检查
 */
export async function authenticateCloudAgent(
  serverId: string,
  agentToken: string
): Promise<CloudAuthResult> {
  // 1. 验证 agentToken
  const server = await getServerByToken(agentToken);
  if (!server) {
    throw new Error('Invalid agent token');
  }

  // 2. 检查服务器是否属于有效的 Tenant
  if (!server.tenantId) {
    throw new Error('Server not associated with any tenant');
  }

  const tenant = await getTenant(server.tenantId);
  if (!tenant) {
    throw new Error('Tenant not found');
  }

  // 3. 检查 Tenant 订阅状态
  if (tenant.plan === 'free' && tenant.subscriptionStatus === 'past_due') {
    throw new Error('Subscription expired. Please update billing.');
  }

  // 4. 检查服务器数量限制
  const serverCount = await getServerCountByTenant(tenant.id);
  if (serverCount > tenant.maxServers) {
    throw new Error(`Server limit exceeded (${tenant.maxServers} max)`);
  }

  return {
    server,
    tenant,
    userId: server.userId,
    permissions: await getUserPermissions(server.userId, tenant.id),
  };
}
```

/**
 * API 客户端（自动注入 tenantId）
 */
export async function apiRequest(
  endpoint: string,
  options?: RequestInit
): Promise<Response> {
  const token = localStorage.getItem('auth_token');
  const tenantId = localStorage.getItem('tenant_id'); // 登录时保存

  const headers = new Headers(options?.headers);
  headers.set('Authorization', `Bearer ${token}`);
  headers.set('X-Tenant-ID', tenantId); // 自动注入

  return fetch(`${API_BASE_URL}${endpoint}`, {
    ...options,
    headers,
  });
}

// packages/cloud/src/api/middleware/verify-tenant.ts

export function verifyTenant() {
  return async (c: Context, next: Next) => {
    const { tenantId: authTenantId } = c.get('auth'); // JWT 中的 tenantId
    const headerTenantId = c.req.header('X-Tenant-ID'); // Header 中的 tenantId

    // 防止用户伪造 tenantId
    if (headerTenantId && headerTenantId !== authTenantId) {
      return c.json({
        error: {
          code: 'TENANT_MISMATCH',
          message: 'Tenant ID mismatch',
        },
      }, 403);
    }

    await next();
  };
}
```

## 🏗️ 技术架构设计

### PostgreSQL vs SQLite

| 场景 | 数据库 | 原因 |
|------|--------|------|
| **Self-Hosted** | SQLite | 单租户，简单部署 |
| **Cloud** | PostgreSQL | 多租户，高并发，备份 |

### 数据库 Schema（Cloud 专属表）

```sql
-- Tenants 表
CREATE TABLE tenants (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  slug VARCHAR(255) UNIQUE NOT NULL, -- 用于子域名
  owner_id INTEGER REFERENCES users(id),
  plan VARCHAR(50) NOT NULL DEFAULT 'free', -- free/pro/team/enterprise
  max_servers INTEGER NOT NULL DEFAULT 1,
  max_users INTEGER NOT NULL DEFAULT 1,
  subscription_status VARCHAR(50), -- active/past_due/canceled
  created_at BIGINT NOT NULL,
  updated_at BIGINT NOT NULL
);

-- Subscriptions 表
CREATE TABLE subscriptions (
  id SERIAL PRIMARY KEY,
  tenant_id INTEGER NOT NULL REFERENCES tenants(id),
  user_id INTEGER NOT NULL REFERENCES users(id),
  plan VARCHAR(50) NOT NULL,
  status VARCHAR(50) NOT NULL, -- incomplete/active/past_due/canceled/unpaid
  stripe_subscription_id VARCHAR(255) UNIQUE,
  stripe_customer_id VARCHAR(255),
  current_period_start BIGINT,
  current_period_end BIGINT,
  cancel_at_period_end BOOLEAN DEFAULT false,
  created_at BIGINT NOT NULL,
  updated_at BIGINT NOT NULL
);

-- AI Usage 表
CREATE TABLE ai_usage (
  id SERIAL PRIMARY KEY,
  user_id INTEGER NOT NULL REFERENCES users(id),
  tenant_id INTEGER NOT NULL REFERENCES tenants(id),
  model VARCHAR(100) NOT NULL, -- haiku/sonnet/opus
  input_tokens INTEGER NOT NULL,
  output_tokens INTEGER NOT NULL,
  cost DECIMAL(10, 6) NOT NULL,
  timestamp BIGINT NOT NULL,
  INDEX idx_tenant_timestamp (tenant_id, timestamp),
  INDEX idx_user_timestamp (user_id, timestamp)
);

-- Skills Executions 表
CREATE TABLE skill_executions (
  id SERIAL PRIMARY KEY,
  user_id INTEGER NOT NULL REFERENCES users(id),
  tenant_id INTEGER NOT NULL REFERENCES tenants(id),
  server_id INTEGER NOT NULL REFERENCES servers(id),
  skill_name VARCHAR(100) NOT NULL,
  status VARCHAR(50) NOT NULL, -- success/failed
  report JSONB, -- AI 分析结果
  duration INTEGER, -- 执行耗时（ms）
  timestamp BIGINT NOT NULL

```
