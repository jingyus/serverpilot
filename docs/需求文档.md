// 服务端核心接口
interface InstallServer {
  // 创建安装会话
  createSession(software: string, userContext: UserContext): SessionId
  
  // 接收客户端环境信息，生成初始安装计划
  analyzeEnvironment(sessionId: SessionId, env: EnvironmentInfo): InstallPlan
  
  // 接收执行结果，决定下一步
  handleStepResult(sessionId: SessionId, result: StepResult): NextAction
  
  // 错误诊断和修复建议
  diagnoseError(sessionId: SessionId, error: ErrorContext): FixStrategy[]
}

// 安装计划
interface InstallPlan {
  steps: InstallStep[]
  estimatedTime: number
  risks: Risk[]
}

// 单个安装步骤
interface InstallStep {
  id: string
  description: string          // 用户可读的说明
  command: string              // 要执行的命令
  expectedOutput?: string      // 预期输出特征
  timeout: number
  canRollback: boolean
  onError: ErrorHandlingStrategy
}

// 知识库结构
interface KnowledgeBase {
  software: {
    openclaw: {
      docs: string[]           // 官方文档
      installGuides: InstallGuide[]
      commonIssues: Issue[]
      dependencies: Dependency[]
    }
  }
  solutions: Map<ErrorSignature, Solution[]>
  successCases: InstallSession[]  // 成功案例学习
}

// 客户端核心接口
interface InstallAgent {
  // 环境探测
  detectEnvironment(): EnvironmentInfo

  // 创建快照（用于回滚）
  createSnapshot(): SnapshotId
  rollback(snapshotId: SnapshotId): void

// 环境信息

  os: {
    platform: 'darwin' | 'linux' | 'win32'
    version: string
    arch: string
  }

    type: 'bash' | 'zsh' | 'fish' | 'powershell'
    version: string
  }
  runtime: {
    node?: string
    python?: string
    // ...
  }
  packageManagers: {
    npm?: string
    pnpm?: string
    brew?: string
    apt?: string
    // ...
  }
  network: {
    canAccessNpm: boolean
    canAccessGithub: boolean
    // ...
  }
  permissions: {
    hasSudo: boolean
    canWriteTo: string[]
  }
}

  stderr: string

// WebSocket 消息协议
type Message = 
  | { type: 'session.create', payload: { software: string } }
  | { type: 'env.report', payload: EnvironmentInfo }
  | { type: 'plan.receive', payload: InstallPlan }
  | { type: 'step.execute', payload: InstallStep }
  | { type: 'step.output', payload: { stepId: string, output: string } }
  | { type: 'step.complete', payload: StepResult }
  | { type: 'error.occurred', payload: ErrorContext }

}

export function saveJsonFile(pathname: string, data: unknown) {
  const dir = path.dirname(pathname);
  fs.mkdirSync(dir, { recursive: true, mode: 0o700 });
  fs.writeFileSync(pathname, `${JSON.stringify(data, null, 2)}\n`, "utf8");
  fs.chmodSync(pathname, 0o600); // 安全权限
}
